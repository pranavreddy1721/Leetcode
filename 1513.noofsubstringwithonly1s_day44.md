class Solution {
public:
    int numSub(string s) {
        const long long MOD = 1e9 + 7;
        long long cnt = 0, result = 0;
        
        for (char c : s) {
            if (c == '1') {
                cnt++;  // count consecutive ones
            } else {
                result = (result + (cnt * (cnt + 1) / 2) % MOD) % MOD;
                cnt = 0;
            }
        }
        
        // Add the last segment if it ends with '1'
        result = (result + (cnt * (cnt + 1) / 2) % MOD) % MOD;
        
        return result;
    }
};
